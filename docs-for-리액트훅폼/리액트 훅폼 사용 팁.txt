1. 여러 필드 한번에 감시 (useWatch)
// 단일 필드
const username = useWatch({ control, name: 'username' })

// 여러 필드 동시 감시
const [name, email, age] = useWatch({ 
  control, 
  name: ['name', 'email', 'age'] 
})

// 전체 폼 감시
const formValues = useWatch({ control })
2. 조건부 필드 초기화 (reset 부분만)
const { reset, setValue } = useForm()

// 특정 필드만 초기화
reset({ menuName: '새 이름' }, { 
  keepDirty: true,  // dirty 상태 유지
  keepTouched: true // touched 상태 유지
})

// 필드 하나만 초기화
setValue('menuName', '새 이름', { 
  shouldDirty: false,  // dirty로 표시 안 함
  shouldValidate: true // 검증 실행
})
3. 동적 에러 메시지 설정
const { setError, clearErrors } = useForm()

// 수동 에러 설정 (API 검증 실패 시 유용)
setError('username', { 
  type: 'manual',
  message: '이미 사용 중인 아이디입니다' 
})

// 에러 제거
clearErrors('username')

// 여러 필드 에러 설정
setError('root.serverError', { 
  type: '500',
  message: '서버 에러가 발생했습니다' 
})
4. 폼 상태 체크 (formState)
const { formState } = useForm()

formState.isDirty        // 폼이 변경되었는지
formState.dirtyFields    // 변경된 필드 목록 { menuName: true }
formState.touchedFields  // 터치된 필드 목록
formState.errors         // 에러 목록
formState.isSubmitting   // 제출 중인지
formState.isValid        // 유효한지
formState.submitCount    // 제출 시도 횟수

// 특정 필드만 체크
if (formState.dirtyFields.menuName) {
  console.log('메뉴명이 변경되었습니다')
}

6. 조건부 검증 (의존성 검증)
<Controller
  name="password"
  control={control}
  rules={{
    required: '비밀번호를 입력하세요',
    minLength: { value: 8, message: '8자 이상' }
  }}
  render={({ field }) => <input {...field} type="password" />}
/>

<Controller
  name="passwordConfirm"
  control={control}
  rules={{
    validate: (value) => {
      const password = watch('password')
      return value === password || '비밀번호가 일치하지 않습니다'
    }
  }}
  render={({ field }) => <input {...field} type="password" />}
/>
7. 배열 필드 관리 (useFieldArray)
import { useFieldArray } from 'react-hook-form'

const { fields, append, remove, insert } = useFieldArray({
  control,
  name: 'items'
})

// 항목 추가
append({ name: '', price: 0 })

// 항목 삭제
remove(index)

// 중간에 삽입
insert(1, { name: '새 항목', price: 100 })

// 렌더링
{fields.map((field, index) => (
  <div key={field.id}>
    <input {...register(`items.${index}.name`)} />
    <button onClick={() => remove(index)}>삭제</button>
  </div>
))}

12. getValues vs watch 차이
const { watch, getValues } = useForm()

// watch: 리렌더링 발생 (실시간 UI 반영 필요 시)
const username = watch('username')

// getValues: 리렌더링 없음 (제출 시 값만 필요할 때)
const handleClick = () => {
  const currentValue = getValues('username')
  console.log(currentValue)
}
